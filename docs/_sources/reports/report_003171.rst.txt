SQL: 3171
---------

.. code-block:: sql
  :linenos:

    SELECT
      issues.branchcode,
      issues.itemnumber,
      issues.date_due
    FROM
      issues
    WHERE
      issues.date_due BETWEEN <<Between the start of the day on date1|date>>  AND (<<the end of the day on date2|date>> + interval 1 day) AND
      issues.branchcode LIKE <<Select library|ZBRAN>>
    GROUP BY
      issues.branchcode,
      issues.itemnumber
    ORDER BY
      issues.branchcode,
      issues.date_due
