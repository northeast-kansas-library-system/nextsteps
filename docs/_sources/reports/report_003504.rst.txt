.. code-block:: sql
  :linenos:

    SELECT
      borrowers.branchcode,
      LEFT(borrowers.zipcode, 5) as zipcode,
      Count(borrowers.borrowernumber) AS Count_borrowernumber
    FROM
      borrowers
    WHERE
      borrowers.branchcode LIKE <<Choose a library|LBRANCH>>
    GROUP BY
      borrowers.branchcode,
      LEFT(borrowers.zipcode, 5)
    ORDER BY
      borrowers.branchcode,
      LEFT(borrowers.zipcode, 5)
