Created on: 2021-07-28 15:14:53

Created by: gwilliams

Notes:

Lists items with a specific "Not for loan" status

  • Shows items that currently have a "Not for loan" status
  • at the library you specify
  • grouped by biblio.biblionumber and items.itemnumber
  • sorted by the standard Next Search Catalog classification scheme
  • links to the bibliographic record for the item

Notes:

Not for loan statuses include:

  • Download or Online Access (5)
  • In Processing (7)
  • Migration (3)
  • Not For Loan (1)
  • Online Access (6)
  • Ordered (-1)
  • Paola processing (-9)
  • Paola repair/mending (-8)
  • Staff Collection (2)

  • 1307 - Not For Loan Items
  • 1407 - Not For Loan Status
  • 1412 - All items in a specific not for loan status

 1  SELECT
 2    CONCAT(
 3      '<a class="btn btn-default btn-xs noprint" ',
 4      'href=\"/cgi-bin/koha/catalogue/detail.pl?biblionumber=',
 5      biblio.biblionumber, '
 6      " target="_blank">Go to title</a>'
 7    ) AS LINK,
 8    items.homebranch,
 9    If(
10      items.permanent_location = items.location,
11      permlocs.lib,
12      Concat(permlocs.lib, ' (', locs.lib, ')')
13    ) AS LOCATION,
14    itypes.description AS ITYPE,
15    ccodes.lib AS CCODE,
16    items.itemcallnumber,
17    biblio.author,
18    Concat_Ws(" ",
19      biblio.title,
20      ExtractValue(biblio_metadata.metadata, '//datafield[@tag="245"]/subfield[@code="h"]'),
21      ExtractValue(biblio_metadata.metadata, '//datafield[@tag="245"]/subfield[@code="b"]'),
22      ExtractValue(biblio_metadata.metadata, '//datafield[@tag="245"]/subfield[@code="n"]'),
23      ExtractValue(biblio_metadata.metadata, '//datafield[@tag="245"]/subfield[@code="p"]'),
24      ExtractValue(biblio_metadata.metadata, '//datafield[@tag="245"]/subfield[@code="c"]')
25    ) AS FULL_TITLE,
26    items.barcode,
27    not_loans.lib AS NOT_FOR_LOAN,
28    items.dateaccessioned AS DATE_ADDED,
29    items.onloan AS DATE_DUE
30  FROM
31    biblio JOIN
32    biblio_metadata ON biblio_metadata.biblionumber = biblio.biblionumber
33     JOIN
34    items ON items.biblionumber = biblio.biblionumber LEFT JOIN
35    (SELECT
36        authorised_values.category,
37        authorised_values.authorised_value,
38        authorised_values.lib
39      FROM
40        authorised_values
41      WHERE
42        authorised_values.category = 'LOC') permlocs ON
43        permlocs.authorised_value = items.permanent_location LEFT JOIN
44    (SELECT
45        authorised_values.category,
46        authorised_values.authorised_value,
47        authorised_values.lib
48      FROM
49        authorised_values
50      WHERE
51        authorised_values.category = 'LOC') locs ON locs.authorised_value =
52        items.location LEFT JOIN
53    (SELECT
54        itemtypes.itemtype,
55        itemtypes.description
56      FROM
57        itemtypes) itypes ON itypes.itemtype = items.itype LEFT JOIN
58    (SELECT
59        authorised_values.category,
60        authorised_values.authorised_value,
61        authorised_values.lib
62      FROM
63        authorised_values
64      WHERE
65        authorised_values.category = 'ccode') ccodes ON ccodes.authorised_value =
66        items.ccode LEFT JOIN
67    (SELECT
68        authorised_values.category,
69        authorised_values.authorised_value,
70        authorised_values.lib
71      FROM
72        authorised_values
73      WHERE
74        authorised_values.category = 'NOT_LOAN') not_loans ON
75        not_loans.authorised_value = items.notforloan
76  WHERE
77    items.homebranch LIKE <<Choose your library|LBRANCH>> AND
78    (items.notforloan LIKE <<Not for loan status|NOT_LOAN:all>> AND
79     items.notforloan <> '')
80  GROUP BY
81    biblio.biblionumber,
82    items.itemnumber
83  ORDER BY
84    items.homebranch,
85    permlocs.lib,
86    ITYPE,
87    CCODE,
88    items.itemcallnumber,
89    biblio.author,
90    FULL_TITLE,
