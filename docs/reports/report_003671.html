Created by: gwilliams

Notes:

Counts the items that are still in transit and orders them by the week they were shipped

  • Shows data for items that are currently in transit
  • at all Next libraries
  • grouped and sorted by the week shipped
  • links to report 3672

Notes:

Created to isolate possible courier issues.

#transit #courier

 2    weeks.START_END AS WEEK_SHIPPED,
 3    all_shipped.COUNT AS SHIPPED_TOTAL,
 4    shipped_and_received.COUNT AS SHIPPED_AND_RECEIVED,
 5    shipped_then_cancelled.COUNT AS SHIPPED_CNX,
 6    Coalesce(shipped_but_not_yet_received.COUNT, 0) AS STILL_IN_TRANSIT,
 7    Concat(
 8      '<a class="btn btn-default btn-xs noprint"',

14      '&param_name=and+%28YYYY-MM-DD%29',
15      '&sql_params=',

19  FROM
20    (
21      SELECT
22        Concat_Ws('',
23          Str_To_Date(Concat(YearWeek(branchtransfers.datesent, 0), ' Sunday'), '%X%V %W'),
24          ' - ',
25          (Str_To_Date(Concat(YearWeek(branchtransfers.datesent + Interval 7 DAY, 0), ' Sunday'), '%X%V %W') - Interval 1 DAY)
26        ) AS START_END,
27        YearWeek(branchtransfers.datesent, 0) AS YEARWEEK,
28        Concat_Ws('',
29          Str_To_Date(Concat(YearWeek(branchtransfers.datesent, 0), ' Sunday'), '%X%V %W')
30        ) AS d1,
31        Concat_Ws('',
32          Str_To_Date(Concat(YearWeek(branchtransfers.datesent + Interval 7 DAY, 0), ' Sunday'), '%X%V %W') - Interval 1 DAY
33        ) AS d2
34      FROM branchtransfers
35      WHERE branchtransfers.datesent < Date_Sub(Date(Now()), Interval DayOfWeek(Now()) + 6 DAY)
36      GROUP BY YearWeek(branchtransfers.datesent, 0)
37    ) weeks
38  LEFT JOIN
39    (
40      SELECT
41        YearWeek(branchtransfers.datesent, 0) AS SHIPPED_WEEK,
42        Count(DISTINCT branchtransfers.branchtransfer_id) AS COUNT
43      FROM branchtransfers
44      GROUP BY YearWeek(branchtransfers.datesent, 0)
45    ) all_shipped ON all_shipped.SHIPPED_WEEK = weeks.YEARWEEK
46  LEFT JOIN
47    (
48      SELECT
49        YearWeek(branchtransfers.datesent, 0) AS SHIPPED_WEEK,
50        Count(DISTINCT branchtransfers.branchtransfer_id) AS COUNT
51      FROM branchtransfers
52      WHERE branchtransfers.datecancelled IS NOT NULL
53      GROUP BY YearWeek(branchtransfers.datesent, 0)
54    ) shipped_then_cancelled ON shipped_then_cancelled.SHIPPED_WEEK = weeks.YEARWEEK
55  LEFT JOIN
56    (
57      SELECT
58        YearWeek(branchtransfers.datesent, 0) AS SHIPPED_WEEK,
59        Count(DISTINCT branchtransfers.branchtransfer_id) AS COUNT
60      FROM branchtransfers
61      WHERE branchtransfers.datearrived IS NOT NULL
62      GROUP BY YearWeek(branchtransfers.datesent, 0)
63    ) shipped_and_received ON shipped_and_received.SHIPPED_WEEK = weeks.YEARWEEK
64  LEFT JOIN
65    (
66      SELECT
67        YearWeek(branchtransfers.datesent, 0) AS SHIPPED_WEEK,
68        Count(DISTINCT branchtransfers.branchtransfer_id) AS COUNT
69      FROM branchtransfers
70      WHERE
71        branchtransfers.datecancelled IS NULL AND
72        branchtransfers.datearrived IS NULL
73      GROUP BY YearWeek(branchtransfers.datesent, 0)
74    ) shipped_but_not_yet_received ON shipped_but_not_yet_received.SHIPPED_WEEK = weeks.YEARWEEK
75  WHERE
76    Coalesce(shipped_but_not_yet_received.COUNT, 0) > 0
77  GROUP BY
78    weeks.START_END,
79    all_shipped.COUNT,
80    shipped_and_received.COUNT,
81    shipped_then_cancelled.COUNT,
82    Coalesce(shipped_but_not_yet_received.COUNT, 0),
83    weeks.YEARWEEK
84  ORDER BY
85    weeks.YEARWEEK DESC
