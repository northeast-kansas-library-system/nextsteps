 6    ) AS LINK_TO_PATRON,
 7    borrowers.borrowernumber,
 8    borrowers.cardnumber,
 9    borrowers.branchcode,
10    borrowers.categorycode,
11    borrowers.dateenrolled,
12    borrowers.dateexpiry,
13    If(
14      Day(Now()) >= 15,
15      Date_Format(Now() + INTERVAL 15 MONTH, '%Y-%m-15'),
16      Date_Format(Now() + INTERVAL 14 MONTH, '%Y-%m-15')
17    ) AS PROJECTED_DELETION,
18    Coalesce(accountlinesx.DUE_SUM, 0) AS AMT_DUE,
19    Coalesce(issuesx.ICOUNT, 0) AS CHECKOUTS,
20    Coalesce(guaranteesx.GCOUNT, 0) AS GUARANTEES,
21    Coalesce(requestsx.Count_reserve_id, 0) AS REQUESTS,
22    Coalesce(expired_attribute.attribute, 0) AS ATTRIBUTE_VALUE,
23    expired_attribute.lib AS ATTRIBUTE
24  FROM
25    borrowers LEFT JOIN
26    (SELECT
27        accountlines.borrowernumber,
28        Format(Sum(accountlines.amountoutstanding), 2) AS DUE_SUM
29      FROM
30        accountlines
31      GROUP BY
32        accountlines.borrowernumber) accountlinesx ON borrowers.borrowernumber =
33        accountlinesx.borrowernumber LEFT JOIN
34    (SELECT
35        issues.borrowernumber,
36        Count(issues.issue_id) AS ICOUNT
37      FROM
38        issues
39      GROUP BY
40        issues.borrowernumber) issuesx ON borrowers.borrowernumber =
41        issuesx.borrowernumber LEFT JOIN
42    (SELECT
43        reserves.borrowernumber,
44        Count(reserves.reserve_id) AS Count_reserve_id
45      FROM
46        reserves
47      GROUP BY
48        reserves.borrowernumber) requestsx ON borrowers.borrowernumber =
49        requestsx.borrowernumber LEFT JOIN
50    (SELECT
51        borrower_attributes.borrowernumber,
52        borrower_attributes.code,
53        borrower_attributes.attribute,
54        authorised_values.lib,
55        authorised_values.category
56      FROM
57        borrower_attributes JOIN
58        authorised_values ON borrower_attributes.attribute =
59            authorised_values.authorised_value
60      WHERE
61        borrower_attributes.code = 'expired' AND
62        authorised_values.category = 'expired') expired_attribute ON
63        borrowers.borrowernumber = expired_attribute.borrowernumber LEFT JOIN
64    (SELECT
65        borrower_relationships.guarantor_id,
66        Count(borrower_relationships.guarantee_id) AS GCOUNT
67      FROM
68        borrower_relationships
69      GROUP BY
70        borrower_relationships.guarantor_id) guaranteesx ON
71        guaranteesx.guarantor_id = borrowers.borrowernumber
72  WHERE
73    borrowers.dateexpiry BETWEEN CurDate() - INTERVAL 1095 DAY AND CurDate() - INTERVAL 730.5 DAY AND
74    borrowers.branchcode LIKE '%' AND
75    Coalesce(borrowers.othernames, "0") NOT LIKE "%SIP%" AND

84  GROUP BY
