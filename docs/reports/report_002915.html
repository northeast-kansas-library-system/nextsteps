Report 002915 contents // GHW - Flexible Item Status Report

Report technical information: 2915

ID number: 2915

Name: GHW - Flexible Item Status Report

Group: Catalog Records and Items

Sub-group: Shelf Lists

Created on: 2017-03-01 09:17:47

Modified on: 2021-04-13 17:10:35

Created by: gwilliams

This file generated: 2023-03-24 16:49:10

Notes:

Generates a list of items based on their status

  • Shows items currently in the system
  • At the library you specify
  • grouped and sorted by the normal Next Search Catalog classification scheme
  • contains links to the item's bibliographic record

Notes:

Replaces the following reports:

  • 1420 - Items with a Lost Status

Click here to run in a new window

SQL: 2915

  1  SELECT
  2    Concat(
  3      '<a href=\"/cgi-bin/koha/catalogue/detail.pl?biblionumber=',
  4      biblio.biblionumber,
  5      '\" target="_blank">',
  6      biblio.biblionumber,
  7      '</a>'
  8    ) AS LINK_TO_TITLE,
  9    items.itemnumber AS itemnumber,
 10    Concat("-", items.barcode, "-") AS BARCODE,
 11    items.homebranch,
 12    items.holdingbranch,
 13    permlocs.lib AS PERM_LOCATION,
 14    locs.lib AS LOCATION,
 15    itypes.description AS ITYPE,
 16    CCODES.lib AS CCODE,
 17    items.itemcallnumber,
 18    biblio.author,
 19    Concat_Ws(' ',
 20      biblio.title,
 21      ExtractValue(biblio_metadata.metadata, '//datafield[@tag="245"]/subfield[@code="b"]'),
 22      ExtractValue(biblio_metadata.metadata, '//datafield[@tag="245"]/subfield[@code="p"]'),
 23      ExtractValue(biblio_metadata.metadata, '//datafield[@tag="245"]/subfield[@code="n"]')
 24    ) AS FULL_TITLE,
 25    items.copynumber,
 26    items.onloan,
 27    items.datelastborrowed,
 28    items.datelastseen,
 29    Coalesce(withdrawn.lib, "-") AS WITHDRAWN_STATUS,
 30    items.withdrawn_on,
 31    Coalesce(lost.lib, "-") AS LOST_STATUS,
 32    items.itemlost_on,
 33    Coalesce(damaged.lib, "-") AS DAMAGED_STATUS,
 34    Coalesce(not_for_loan.lib, "-") AS NOT_FOR_LOAN_STATUS,
 35    items.replacementprice,
 36    If(paidfor.amountoutstanding = 0, "$0.00 outstanding lost fees", "") AS OUTSTANDING_FEES,
 37    not_for_loan.lib_opac
 38  FROM
 39    items JOIN
 40    biblio ON items.biblionumber = biblio.biblionumber JOIN
 41    biblio_metadata ON biblio_metadata.biblionumber = biblio.biblionumber AND
 42        items.biblionumber = biblio_metadata.biblionumber JOIN
 43    authorised_values ccode ON items.ccode = ccode.authorised_value LEFT JOIN
 44    (SELECT
 45        authorised_values.authorised_value,
 46        authorised_values.lib,
 47        authorised_values.category,
 48        authorised_values.lib_opac
 49      FROM
 50        authorised_values
 51      WHERE
 52        authorised_values.category = 'NOT_LOAN') not_for_loan ON
 53        items.notforloan = not_for_loan.authorised_value LEFT JOIN
 54    (SELECT
 55        authorised_values.authorised_value,
 56        authorised_values.lib,
 57        authorised_values.category,
 58        authorised_values.lib_opac
 59      FROM
 60        authorised_values
 61      WHERE
 62        authorised_values.category = 'WITHDRAWN') withdrawn ON
 63        items.withdrawn = withdrawn.authorised_value LEFT JOIN
 64    (SELECT
 65        authorised_values.authorised_value,
 66        authorised_values.lib,
 67        authorised_values.category,
 68        authorised_values.lib_opac
 69      FROM
 70        authorised_values
 71      WHERE
 72        authorised_values.category = 'LOST') lost ON items.itemlost =
 73        lost.authorised_value LEFT JOIN
 74    (SELECT
 75        authorised_values.authorised_value,
 76        authorised_values.lib,
 77        authorised_values.category,
 78        authorised_values.lib_opac
 79      FROM
 80        authorised_values
 81      WHERE
 82        authorised_values.category = 'DAMAGED') damaged ON items.damaged =
 83        damaged.authorised_value LEFT JOIN
 84    (SELECT
 85        itemtypes.itemtype,
 86        itemtypes.description
 87      FROM
 88        itemtypes) itypes ON itypes.itemtype = items.itype LEFT JOIN
 89    (SELECT
 90        accountlines.itemnumber,
 91        accountlines.amountoutstanding,
 92        accountlines.note,
 93        accountlines.credit_type_code,
 94        accountlines.debit_type_code
 95      FROM
 96        accountlines
 97      WHERE
 98        accountlines.amountoutstanding = 0 AND
 99        accountlines.debit_type_code LIKE 'L%') paidfor ON paidfor.itemnumber =
100        items.itemnumber LEFT JOIN
101    (SELECT
102        authorised_values.category,
103        authorised_values.authorised_value,
104        authorised_values.lib,
105        authorised_values.lib_opac
106      FROM
107        authorised_values
108      WHERE
109        authorised_values.category = 'LOC') permlocs ON
110        permlocs.authorised_value = items.permanent_location LEFT JOIN
111    (SELECT
112        authorised_values.category,
113        authorised_values.authorised_value,
114        authorised_values.lib,
115        authorised_values.lib_opac
116      FROM
117        authorised_values) locs ON locs.authorised_value = items.location
118    LEFT JOIN
119    (SELECT
120        authorised_values.category,
121        authorised_values.authorised_value,
122        authorised_values.lib,
123        authorised_values.lib_opac
124      FROM
125        authorised_values
126      WHERE
127        authorised_values.category = 'CCODE') CCODES ON CCODES.authorised_value =
128        items.ccode
129  WHERE
130    items.homebranch LIKE <<Select library|ZBRAN>> AND
131    Coalesce(lost.lib, "-") LIKE <<Lost status|ZLOST>> AND
132    items.damaged LIKE <<Damaged status|ZDAMAGED>> AND
133    items.withdrawn LIKE <<Withdrawn status|ZWITHDRAWN>> AND
134    Coalesce(not_for_loan.lib_opac, "-") LIKE <<Not-for-loan status|LNOT_LOAN>>
135  GROUP BY
136    items.itemnumber,
137    CCODES.lib,
138    not_for_loan.lib_opac,
139    biblio.biblionumber
140  ORDER BY
141    items.homebranch,
142    PERM_LOCATION,
143    ITYPE,
144    CCODE,
145    items.itemcallnumber,
146    biblio.author,
147    FULL_TITLE
