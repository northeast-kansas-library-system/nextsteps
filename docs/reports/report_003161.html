Report 003161 contents // GHW - Patrons added in the previous calendar month

Report technical information: 3161

ID number: 3161

Name: GHW - Patrons added in the previous calendar month

Created on: 2019-01-27 23:01:25

Modified on: 2022-10-05 16:04:27

Cache expiry in seconds: 300

Created by: gwilliams

This file generated: 2023-03-24 16:49:10

Notes:

Generates a list of new patrons to review for input errors

  • Shows patrons added in the previous calendar month
  • at the library you specify
  • grouped by borrowernumber
  • sorted by the patron's last name and first name
  • contains links to the edit patron screen for each patron

Notes:

Partially replaces report 555

Click here to run in a new window

SQL: 3161

 1  SELECT
 2    Concat(
 3      '<a class="btn btn-default" href="/cgi-bin/koha/members/memberentry.pl?op=modify&borrowernumber=',
 4      borrowers.borrowernumber,
 5      '\" target="_blank">Edit borrower account</a>'
 6    ) AS EDIT,
 7    Concat_Ws(' ',
 8      borrowers.surname,
 9      '/',
10      borrowers.firstname,
11      If(
12        borrowers.othernames = '',
13        '',
14        Concat('(', borrowers.othernames, ')')
15      ),
16      IF(borrowers.dateofbirth IS NULL, '', Concat('<br /><span style="text-decoration: underline;">Birthdate:</span> ', borrowers.dateofbirth))
17    ) AS IDENTITY,
18    Concat_Ws('<br />',
19      Concat(borrowers.address, If(borrowers.address2 = '', '', Concat('<br />', borrowers.address2))),
20      Concat(
21        If(borrowers.city = '', '', Concat(borrowers.city, ', ')),
22        If(borrowers.state = '', '', Concat(borrowers.state, ' ')),
23        If(borrowers.zipcode = '', '', borrowers.zipcode)
24      )
25    ) AS MAIN_ADDRESS,
26    Concat_Ws('',
27      If(borrowers.phone = '', '', Concat('<i class="fa fa-phone-square fa-lg" aria-hidden="true" style="color: green;"></i> ', borrowers.phone, '<br />')),
28      If(borrowers.phonepro = '', '', Concat('<i class="fa fa-phone-square" aria-hidden="true" style="color: green;"></i> ', borrowers.phonepro, '<br />')),
29      IF(borrowers.email = '', '', CONCAT('<i class="fa fa-envelope fa-lg" aria-hidden="true" style="color: blue;"></i> ', borrowers.email, '<br />')),
30      IF(borrowers.emailpro = '', '', CONCAT('<i class="fa fa-envelope" aria-hidden="true" style="color: blue;"></i> ', borrowers.emailpro, '<br />'))
31    ) AS CONTACT_INFO,
32    CONCAT_WS('<br />',
33      Concat('Card number: ', borrowers.cardnumber),
34      Concat('Library: ', branches.branchname),
35      Concat('Category: ', categories.description)
36    ) AS LIBRARY_MANAGEMENT,
37    CONCAT_WS('<br />',
38      Concat('Registration date: ', borrowers.dateenrolled),
39      Concat('Expiration date: ', borrowers.dateexpiry)
40    ) AS LIBRARY_SETUP,
41    If(borrowers.smsalertnumber = '', '', Concat('Provider: ', sms_providers.name, '<br />SMS number: ', borrowers.smsalertnumber)) AS SMS,
42    @SortOrder := <<Sort by|XS_BORROWER>> AS SORTING
43  FROM
44    borrowers LEFT JOIN
45    sms_providers ON borrowers.sms_provider_id = sms_providers.id LEFT JOIN
46    branches ON borrowers.branchcode = branches.branchcode LEFT JOIN
47    categories ON borrowers.categorycode = categories.categorycode
48  WHERE
49    borrowers.branchcode LIKE <<Choose the patron home library|LBRANCH>> AND
50    Year(borrowers.dateenrolled) = Year(Now() - INTERVAL 1 MONTH) AND
51    Month(borrowers.dateenrolled) = Month(Now() - INTERVAL 1 MONTH)
52  GROUP BY
53    borrowers.borrowernumber
54  ORDER BY
55    (CASE WHEN SORTING = 13 THEN borrowers.dateofbirth END) DESC,
56    (CASE WHEN SORTING = 12 THEN borrowers.dateexpiry END) DESC,
57    (CASE WHEN SORTING = 11 THEN borrowers.dateenrolled END) DESC,
58    (CASE WHEN SORTING = 10 THEN borrowers.surname END) DESC,
59    (CASE WHEN SORTING = 9 THEN borrowers.dateofbirth END) ASC,
60    (CASE WHEN SORTING = 8 THEN borrowers.dateexpiry END) ASC,
61    (CASE WHEN SORTING = 7 THEN borrowers.dateenrolled END) ASC,
62    (CASE WHEN SORTING = 6 THEN borrowers.zipcode END) ASC,
63    (CASE WHEN SORTING = 5 THEN borrowers.city END) ASC,
64    (CASE WHEN SORTING = 4 THEN borrowers.state END) ASC,
65    (CASE WHEN SORTING = 3 THEN borrowers.surname END) ASC,
66    (CASE WHEN SORTING = 2 THEN categories.description END) ASC,
67    (CASE WHEN SORTING = 1 THEN branches.branchname END) ASC,
68    branches.branchname,
69    categories.description,
70    borrowers.surname,
71    borrowers.firstname,
72    borrowers.state,
73    borrowers.city,
74    borrowers.address,
75    borrowers.zipcode,
76    borrowers.dateenrolled,
77    borrowers.dateexpiry,
78    borrowers.dateofbirth
79  LIMIT 1000
