Notes:

Shows turnover rate by location and collection code

  • in the previous 12 months
  • at the library you specify
  • grouped by library, location, and collection code
  • sorted by library, location, and collection code with an "All locations" and "All collection codes" grouping at the top of the results

Notes:

Developed for "Collection development 2.0" workshop

Please note that these calculations include all items checked out at your library regardless of which library owned the item. If, for example, your library doesn't own any items with a "VG-Nintendo 3DS" collection code but one from another library was checked out at your library to fill a request, that item will be counted in these calculations.

"CURRENT_ITEM_COUNT" is based on an item's collection code at the time the report was run. "CKO_RENEW_COUNT" is based on an item's collection code at the time it was checked out. It an item's collection code changes between the time it was checked out and the time this report is run, the results will be skewed. This can be particularly true with temporary items such as ILL items.

592    Concat('------') AS CURRENT_ITEM_COUNT,
593    Concat('------') AS CKO_RENEW_COUNT,
594    Concat('------') AS TURNOVER
595  UNION
596  SELECT

598    branchesitypes.lib AS LOCATION,
599    branchesitypes.lib1 AS CCODE,
600    icounts.Count_itemnumber AS CURRENT_ITEM_COUNT,
601    statcounts.COUNT AS CKO_RENEW_COUNT,
602    round((statcounts.COUNT / icounts.Count_itemnumber), 2) AS TURNOVER
603  FROM
604    (SELECT
605        If(branches.branchcode LIKE 'DONI%', 'DONIPHAN_COMBINED',
606          If(branches.branchcode LIKE 'PH%', 'PH_COMBINED',
607            branches.branchcode)) AS branchcode,
608        biblocs.category,
609        biblocs.authorised_value,
610        biblocs.lib,
611        ccodes.authorised_value AS authorised_value1,
612        ccodes.lib AS lib1
613      FROM
614        branches,
615        (SELECT
616            authorised_values.category,
617            authorised_values.authorised_value,
618            authorised_values.lib
619          FROM
620            authorised_values
621          WHERE
622            authorised_values.category = 'bibloc' AND
623            authorised_values.authorised_value <> "YOUNGADULT") biblocs,
624        (SELECT
625            authorised_values.category,
626            authorised_values.authorised_value,
627            authorised_values.lib
628          FROM
629            authorised_values
630          WHERE
631            authorised_values.category = 'ccode') ccodes) branchesitypes LEFT JOIN
632    (SELECT
633        If(items.homebranch LIKE 'DONI%', 'DONIPHAN_COMBINED', If(items.homebranch
634        LIKE 'PH%', 'PH_COMBINED', items.homebranch)) AS homebranch,
635        If(items.permanent_location = "BALDADULT", "ADULT",
636          If(items.permanent_location = "LVPLADULT", "ADULT",
637            If(items.permanent_location = "PAOLAADULT", "ADULT",
638              If(items.permanent_location LIKE "%CHILD%", "CHILDRENS",
639                If(items.permanent_location LIKE "%Y%", "L_YA",
640                  If(items.permanent_location LIKE "%CA%", "ADULT",
641                    If(items.permanent_location LIKE "%PROC%", "ADULT",
642                      If(items.permanent_location IS NULL, "ADULT",
643                        items.permanent_location
644                      )
645                    )
646                  )
647                )
648              )
649            )
650          )
651        ) AS permanent_location,
652        Coalesce(items.ccode, "XXX") AS ccode,
653        Count(items.itemnumber) AS Count_itemnumber
654      FROM
655        items
656      GROUP BY
657        If(items.homebranch LIKE 'DONI%', 'DONIPHAN_COMBINED',
658          If(items.homebranch LIKE 'PH%', 'PH_COMBINED',
659            items.homebranch)),
660        If(items.permanent_location = "BALDADULT", "ADULT",
661          If(items.permanent_location = "LVPLADULT", "ADULT",
662            If(items.permanent_location = "PAOLAADULT", "ADULT",
663              If(items.permanent_location LIKE "%CHILD%", "CHILDRENS",
664                If(items.permanent_location LIKE "%Y%", "L_YA",
665                  If(items.permanent_location LIKE "%CA%", "ADULT",
666                    If(items.permanent_location LIKE "%PROC%", "ADULT",
667                      If(items.permanent_location IS NULL, "ADULT",
668                        items.permanent_location
669                      )
670                    )
671                  )
672                )
673              )
674            )
675          )
676        ),
677        Coalesce(items.ccode, "XXX")) icounts ON icounts.homebranch =
678        branchesitypes.branchcode AND
679        icounts.permanent_location = branchesitypes.authorised_value AND
680        icounts.ccode = branchesitypes.authorised_value1 LEFT JOIN
681    (SELECT
682        If(statistics.branch LIKE 'DONI%', 'DONIPHAN_COMBINED',
683          If(statistics.branch LIKE 'PH%', 'PH_COMBINED',
684            statistics.branch)) AS
685        branch,
686        If(statistics.location = "BALDADULT", "ADULT",
687          If(statistics.location = "LVPLADULT", "ADULT",
688            If(statistics.location = "PAOLAADULT", "ADULT",
689              If(statistics.location LIKE "%CHILD%", "CHILDRENS",
690                If(statistics.location LIKE "%Y%", "L_YA",
691                  If(statistics.location LIKE "%CA%", "ADULT",
692                    If(statistics.location LIKE "%PROC%", "ADULT",
693                      If(statistics.location IS NULL, "ADULT",
694                        statistics.location
695                      )
696                    )
697                  )
698                )
699              )
700            )
701          )
702        ) AS LOCATION,
703        Coalesce(statistics.ccode, "XXX") AS ccode,
704        count(*) AS COUNT
705      FROM
706        statistics
707      WHERE
708        (statistics.type = 'issue' OR
709          statistics.type = 'renew') AND
710        statistics.datetime BETWEEN
711          (AddDate(Last_Day(SubDate(Now(), INTERVAL 13 MONTH)), 1)) AND
712          (AddDate(Last_Day(SubDate(Now(), INTERVAL 1 MONTH)), 1))
713      GROUP BY
