Report 003039 contents // GHW - Flexible requests history report

Report technical information: 3039

ID number: 3039

Name: GHW - Flexible requests history report

Group: Holds-Reserves

Created on: 2018-01-16 15:28:12

Modified on: 2018-04-16 11:13:46

Cache expiry in seconds: 300

Created by: gwilliams

This file generated: 2023-03-24 16:49:10

Notes:

Generates a screen readable report of current and past request information based on very flexible parameters

  • Shows the history of any active, filled, and cancelled requests in NExpress
  • at all pickup libraries or at the pickup library you specify
  • with the request status and progress that you specify
  • with the hold suspension status you specify
  • and with the library card, biblionumber, or item barcode number you specify
  • grouped by reserve ID number
  • sorted by the last time the request was updated
  • contains links to the action logs, the patron record, the bibliographic record, and the item's in transit history

Notes:

Contains active links to reports 2784 and 3040

Click here to run in a new window

SQL: 3039

 1  SELECT
 2    requests.reserve_id,
 3    Concat("Status: ", requests.statuss, "<br /> Progress: ", If(requests.cancellationdate IS NOT NULL, "Cancelled", If(requests.found = "T", "In transit", If(requests.found = "F", "Filled / finished", If(requests.found = "W", "Waiting for pickup", "Still active"))))) AS STATUS,
 4    Concat("Patron BC: ", borrowers.cardnumber, "<br />", "Pickup at: ", requests.branchcode) AS PATRON_INFO,
 5    Concat_Ws("<br />",
 6      Concat("Requested on: ", requests.reservedate), Concat("Waiting since: ", requests.waitingdate),
 7      Concat("Expires on: ", requests.expirationdate), Concat("Cancelled on: ", requests.cancellationdate),
 8      Concat("Updated on: ", requests.timestamp)) AS DATES,
 9    Concat_Ws("<br />", If(requests.suspend <> 0, "Suspended", "-"), If(requests.suspend = 0, "-", If(requests.suspend_until IS NOT NULL, Concat("until ", Date_Format(requests.suspend_until, "%Y.%m.%d")), "indefinitely"))) AS SUSPENDED,
10    requests.biblionumber AS BIBLIO_NUMBER,
11    Concat_Ws("<br />",
12      Concat("BC: ", Coalesce(Coalesce(items.barcode, "-"), Coalesce(deleteditems.barcode, "-"))),
13      Concat("Home: ", Coalesce(Coalesce(items.homebranch, "-"), Coalesce(deleteditems.homebranch, "-"))),
14      Concat("Location: ", Coalesce(Coalesce(items.location, "-"), Coalesce(deleteditems.location, "-"))),
15      Concat("Type: ", Coalesce(Coalesce(items.itype, "-"), Coalesce(deleteditems.itype, "-"))),
16      Concat("Collection: ", Coalesce(Coalesce(items.ccode, "-"), Coalesce(deleteditems.ccode, "-"))),
17      Concat("Call#: ", Coalesce(Coalesce(items.itemcallnumber, "-"), Coalesce(deleteditems.itemcallnumber, "-"))),
18      Concat("Author: ", Coalesce(Coalesce(biblio.author, "-"), Coalesce(deletedbiblio.author, "-"))),
19      Concat("Title: ", Coalesce(Coalesce(biblio.title, "-"), Coalesce(deletedbiblio.title, "-")))) AS ITEM_INFO,
20    CONCAT_WS("<br />",
21      Concat("Action logs: ", Concat("<a href='/cgi-bin/koha/reports/guided_reports.pl?reports=3040&phase=Run+this+report&sql_params=", requests.reserve_id,"' target='_blank'>see last 60 days of activity</a>")),
22      Concat("Link to patron: ", Concat("<a href='/cgi-bin/koha/circ/circulation.pl?borrowernumber=", requests.borrowernumber,"' target='_blank'>go to the borrower's record</a>")),
23      Concat("Link to title: ", Concat("<a href='/cgi-bin/koha/catalogue/detail.pl?biblionumber=", requests.biblionumber,"' target='_blank'>go to the bibliographic record</a>")),
24      Concat("Link to item: ", Concat("<a href='/cgi-bin/koha/catalogue/moredetail.pl?itemnumber=", items.itemnumber, "&biblionumber=", biblio.biblionumber, "' target='_blank'>go to the item record</a>")),
25      Concat("Item in transit history: ", Concat("<a href='/cgi-bin/koha/reports/guided_reports.pl?reports=2784&phase=Run+this+report&sql_params=", items.barcode,"' target='_blank'>see item transit history</a>"))
26  ) AS LINKS
27  FROM
28    borrowers
29    JOIN (SELECT
30          reserves.reserve_id,
31          reserves.borrowernumber,
32          reserves.reservedate,
33          reserves.biblionumber,
34          reserves.branchcode,
35          reserves.notificationdate,
36          reserves.reminderdate,
37          reserves.cancellationdate,
38          reserves.reservenotes,
39          reserves.priority,
40          reserves.found,
41          reserves.timestamp,
42          reserves.itemnumber,
43          reserves.waitingdate,
44          reserves.expirationdate,
45          reserves.lowestPriority,
46          reserves.suspend,
47          reserves.suspend_until,
48          reserves.itemtype,
49          If(reserves.reserve_id IS NOT NULL, "Active", "-") AS statuss
50        FROM
51          reserves
52        UNION
53        SELECT
54          old_reserves.reserve_id,
55          old_reserves.borrowernumber,
56          old_reserves.reservedate,
57          old_reserves.biblionumber,
58          old_reserves.branchcode,
59          old_reserves.notificationdate,
60          old_reserves.reminderdate,
61          old_reserves.cancellationdate,
62          old_reserves.reservenotes,
63          old_reserves.priority,
64          old_reserves.found,
65          old_reserves.timestamp,
66          old_reserves.itemnumber,
67          old_reserves.waitingdate,
68          old_reserves.expirationdate,
69          old_reserves.lowestPriority,
70          old_reserves.suspend,
71          old_reserves.suspend_until,
72          old_reserves.itemtype,
73          If(old_reserves.cancellationdate IS NOT NULL, "Cancelled", "Filled") AS statuss
74        FROM
75          old_reserves) requests ON borrowers.borrowernumber = requests.borrowernumber
76    LEFT JOIN items ON requests.itemnumber = items.itemnumber
77    LEFT JOIN deleteditems ON requests.itemnumber = deleteditems.itemnumber
78    LEFT JOIN biblio ON requests.biblionumber = biblio.biblionumber
79    LEFT JOIN deletedbiblio ON requests.biblionumber = deletedbiblio.biblionumber
80  WHERE
81    requests.branchcode LIKE <<Choose pickup library|LBRANCH>> AND
82    requests.statuss LIKE <<Choose request status|LHOLDACT>> AND
83    If(requests.cancellationdate IS NOT NULL, "Cancelled", If(requests.found = "T", "In transit", If(requests.found = "F", "Filled", If(requests.found = "W", "Waiting for pickup", "Still active")))) LIKE <<Choose request progress|LHOLDPROG>> AND
84    If(requests.suspend <> 0, "Suspended", "-") LIKE <<Choose suspended status|LHOLDSUS>> AND
85    borrowers.cardnumber LIKE Concat("%", <<Enter library card number or a % symbol>>, "%") AND
86    requests.biblionumber LIKE Concat("%", <<Enter title biblio number or a % symbol>>, "%") AND
87    Coalesce(Coalesce(items.barcode, "-"), Coalesce(deleteditems.barcode, "-")) LIKE Concat("%", <<Enter item barcode number or a % symbol>>, "%")
88  GROUP BY
89    requests.reserve_id
90  ORDER BY
91    requests.timestamp DESC
