    SELECT
      branches.branchcode AS BRANCH,
      Coalesce(TOTAL_PATRONS.COUNT, 0) AS '2.2a_TOTAL_PATRONS'
    FROM
      branches LEFT JOIN
      (SELECT
        borrowers.branchcode,
        Count(*) AS COUNT
      FROM
        borrowers
      GROUP BY
        borrowers.branchcode) TOTAL_PATRONS
        ON branches.branchcode = TOTAL_PATRONS.branchcode
    GROUP BY
      branches.branchcode
