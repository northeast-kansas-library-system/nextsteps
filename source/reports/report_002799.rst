SQL: 2799
---------

.. code-block:: sql
  :linenos:

    SELECT
      items.itemnumber,
      items.barcode,
      items.homebranch,
      items.holdingbranch,
      items.location,
      items.itype,
      items.ccode,
      items.itemcallnumber,
      biblio.author,
      biblio.title,
      items.timestamp
    FROM
      items INNER JOIN
      biblio
        ON items.biblionumber = biblio.biblionumber
    WHERE
      items.homebranch LIKE <<Enter item home branch|LBRANCH>> AND
      items.paidfor IS NOT NULL AND
      items.paidfor <> " "
    ORDER BY
      items.homebranch,
      items.location,
      items.itype,
      items.ccode,
      items.itemcallnumber,
      biblio.author,
      biblio.title
